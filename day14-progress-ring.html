⭕ Circular Progress Ring — SVG, Animated

One circle, one number — progress you can feel

Drag the slider to drive the big ring and watch the number count up. Tap a preset (0 / 25 / 75 / 100), pick a colour, or change the size. It's just two SVG circles and a tiny bit of dash math.

The whole trick: a circle's outline has a length — circumference = 2πr. Make the dash as long as the whole circle, then hide part of it with stroke-dashoffset = C × (1 − percent). Offset shrinks → more arc shows.

🔨 Build it yourself

    
Two SVG circles, one line of dash math, and a CSS transition. ~50 lines, no libraries.

    
    

      
  1. 1
    Two stacked circles — a track and a progress arc
    
        
    <!-- both circles share the SAME centre & radius -->
<svg class="ring" width="220" height="220" viewBox="0 0 220 220">
  <!-- the faint groove behind -->
  <circle class="track" cx="110" cy="110" r="92" stroke-width="16"/>
  <!-- the coloured arc that we'll reveal -->
  <circle class="prog"  cx="110" cy="110" r="92" stroke-width="16"/>
</svg>
    
      
    2. 

      
  2. 2
    Base styling — no fill, round caps, start at the top
    
        
    .ring  { transform: rotate(-90deg); }   /* arc begins at 12 o'clock */
.track { fill:none; stroke:#e2e8f0; }    /* hollow circles only */
.prog  {
  fill:none;
  stroke:#6366f1;
  stroke-linecap:round;                   /* nicely rounded ends */
  transition: stroke-dashoffset .6s cubic-bezier(.4,0,.2,1);
}
    
      
    3. 

      
  3. 3
    Measure the circle: circumference = 2πr
    
        
    const prog = document.querySelector(".prog");
const r = prog.r.baseVal.value;          // 92
const C = 2 * Math.PI * r;               // ≈ 578.05

// make the dash exactly as long as the whole outline,
// so one full dash == the full circle.
prog.style.strokeDasharray = C;
    
      
    4. 

      
  4. 4
    Map percent → dashoffset (the reveal)
    
        
    // offset = how much of the dash to HIDE.
//   percent 0   → hide all (offset = C)      → no arc
//   percent 100 → hide none (offset = 0)     → full ring
function setProgress(percent) {
  const offset = C * (1 - percent / 100);
  prog.style.strokeDashoffset = offset;   // CSS transition animates it
}
setProgress(75);

    6. 

      
  6. 6
    Animate the number in JS to match the arc
    
        
    const num = document.getElementById("num");
let shown = 0;
function countTo(target) {
  const step = () => {
    shown += (target - shown) * 0.15;          // ease toward target
    if (Math.abs(target - shown) < 0.5) shown = target;
    num.textContent = Math.round(shown);
    if (shown !== target) requestAnimationFrame(step);
  };
  step();
}

  7. 7
    Wire up the slider + presets + colour
    
        
    slider.addEventListener("input", e => {
  const p = +e.target.value;
  setProgress(p);      // moves the arc
  countTo(p);          // counts the number
});
presets.forEach(b => b.onclick = () => {
  slider.value = b.dataset.preset;
  slider.dispatchEvent(new Event("input"));
});
colour.oninput = e => prog.style.stroke = e.target.value;

.spin-svg  { animation: spin 1.4s linear infinite; transform-origin:center; }
@keyframes spin { to { transform: rotate(360deg); } }

.spin-prog { fill:none; stroke-linecap:round;
             animation: dash 1.4s ease-in-out infinite; }
@keyframes dash {
  0%   { stroke-dasharray:1,200;  stroke-dashoffset:0;   }
  50%  { stroke-dasharray:90,200; stroke-dashoffset:-35; }
  100% { stroke-dasharray:90,200; stroke-dashoffset:-124;}
}
