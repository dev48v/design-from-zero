    Fuzzy match: chars in order
    // "tgldk" matches "Toggle dark mode"
function fuzzy(q, text){
  let i = 0;
  for (const ch of text.toLowerCase())
    if (ch === q[i]) i++;
  return i === q.length;   // all query chars found, in order
}
