🖱️ Context Menu — Custom Right-Click Menu with Edge-Flip Positioning, Nested Submenus, Full Keyboard Nav & ARIA Roles

Right-click any file — or the empty space — to open the menu.

This is a real context menu built from scratch in vanilla JS. It blocks the browser's native menu (preventDefault on contextmenu) and opens its own at the cursor. It has icons + shortcuts, separators, a disabled item, a nested submenu (hover Share) and a destructive item. It is edge-aware — right-click near the right or bottom edge and it flips so it never spills off-screen. Full keyboard: ↑↓ move, →/← open/close the submenu, Enter activate, Esc close. It also dismisses on outside-click, scroll and window blur.

🔨 Build it yourself

    
A menu described as a plain data array, rendered to a positioned element, dropped at the cursor with a little clamp/flip geometry so it never leaves the viewport, a roving-tabindex keyboard model with a submenu that opens on → and hover, and a few dismiss listeners (outside-click, Esc, scroll, blur) — plus the one line that makes it all possible: preventDefault on the contextmenu event. ~130 lines of vanilla JS, no dependencies.

    
    
      
  1. 
        
    1
    Block the native menu, open yours
    
        
    Listen for contextmenu on the surface and call preventDefault() — that suppresses the browser's own right-click menu. The event carries clientX/clientY: the exact cursor point to open at.
    
        
    surface.addEventListener("contextmenu", e => {
  e.preventDefault();                       // kill the native menu
  const card = e.target.closest(".file-card");
  select(card);                             // remember what was clicked
  openRoot(e.clientX, e.clientY);           // open ours at the cursor
});

    2
    Describe the menu as data
    
        
    Don't hand-write markup for every item. Describe the menu as an array of plain objects — icon, label, shortcut, and flags for sep / disabled / danger / submenu — and render it. Now the structure is data you can reorder, gate or generate.
    
        
    const MENU = [
  { icon:"✂️", label:"Cut",   shortcut:"Ctrl+X" },
  { icon:"📋", label:"Copy",  shortcut:"Ctrl+C" },
  { icon:"📎", label:"Paste", shortcut:"Ctrl+V", disabled:true },
  { sep:true },
  { icon:"📤", label:"Share", submenu:[ /* … */ ] },
  { sep:true },
  { icon:"🗑️", label:"Delete", shortcut:"Del", danger:true },
];
    
      
    3. 
      
  3. 
        
    3
    Render each item with its ARIA role
    
        
    Build a role="menu" container. Separators become role="separator"; every other row is role="menuitem" with tabindex="-1" (roving). Items with a submenu get aria-haspopup + aria-expanded; disabled ones get aria-disabled.
    
        
    function buildMenu(items){
  const el = document.createElement("div");
  el.className = "ctx-menu"; el.setAttribute("role","menu"); el.tabIndex = -1;
  for (const it of items){
    if (it.sep){ el.insertAdjacentHTML("beforeend",
      '<div class="ctx-sep" role="separator"></div>'); continue; }
    const b = document.createElement("div");
    b.setAttribute("role","menuitem"); b.tabIndex = -1;
    b.className = "ctx-item" + (it.danger?" danger":"") + (it.disabled?" disabled":"");
    if (it.disabled) b.setAttribute("aria-disabled","true");
    if (it.submenu){ b.setAttribute("aria-haspopup","true");
                     b.setAttribute("aria-expanded","false"); }
    b._item = it; b.innerHTML = row(it); el.appendChild(b);
  }
  return el;
}
    
      
    4. 
      
  4. 
        
    4
    Place at the cursor — then flip & clamp
    
        
    The whole trick. Measure the menu after it is in the DOM (append hidden, read getBoundingClientRect). If opening at the cursor would overflow the right edge, flip it to open leftward; likewise flip up for the bottom edge. Then clamp so it can never sit off-screen. Because the menu is position:fixed, clientX/clientY map straight to left/top.
    
        
    function placeMenu(x, y, mw, mh, vw, vh, margin){
  margin = margin || 8;
  let px = x;
  if (x + mw + margin > vw) px = x - mw;        // flip left off the right edge
  if (px < margin) px = vw - mw - margin;        // still overflowing? pin right
  if (px < margin) px = margin;                  // menu wider than screen: pin left
  let py = y;
  if (y + mh + margin > vh) py = y - mh;         // flip up off the bottom edge
  if (py < margin) py = vh - mh - margin;
  if (py < margin) py = margin;
  return { x: px, y: py };
}
    
      
    5. 
      
  5. 
        
    5
    Roving tabindex + keyboard nav
    
        
    Only one item is "active" at a time. setActive moves a highlight and calls .focus() on that row (wrapping at the ends). One keydown handler on the deepest open menu drives everything: arrows move, Home/End jump, Enter activates, Esc closes one level.
    
        
    function setActive(rec, i){
  const items = rec.items; if (!items.length) return;
  i = (i + items.length) % items.length;                 // wrap
  items.forEach(el => el.classList.remove("active"));
  rec.active = i; items[i].classList.add("active"); items[i].focus();
}
function onKey(e){
  const rec = stack[stack.length - 1]; if (!rec) return;  // deepest menu
  const k = e.key, cur = rec.items[rec.active];
  if (k === "ArrowDown"){ e.preventDefault(); setActive(rec, rec.active + 1); }
  else if (k === "ArrowUp"){ e.preventDefault(); setActive(rec, rec.active - 1); }
  else if (k === "Home"){ e.preventDefault(); setActive(rec, 0); }
  else if (k === "End"){ e.preventDefault(); setActive(rec, rec.items.length - 1); }
  else if (k === "Enter" || k === " "){ e.preventDefault(); activate(rec, cur); }
  else if (k === "Escape"){ e.preventDefault(); closeTop(); }
  else if (k === "Tab"){ e.preventDefault(); setActive(rec, rec.active + (e.shiftKey?-1:1)); }
}
    
      
    6. 
      
  6. 
        
    6
    Submenus: a stack of menus
    
        
    Keep open menus in a stack. Opening a submenu builds another menu next to the parent item (flipping left if it would overflow) and pushes it — so it becomes the keyboard target. → / hover open it; ← pops it and returns focus to the parent row.
    
        
    function openSub(rec, itemEl){
  const it = itemEl._item; if (!it.submenu) return;
  itemEl.setAttribute("aria-expanded","true");
  const el = buildMenu(it.submenu); layer.appendChild(el);
  const m = el.getBoundingClientRect(), r = itemEl.getBoundingClientRect();
  let x = r.right - 2;                                  // prefer the right side
  if (x + m.width + 8 > innerWidth) x = r.left - m.width + 2;   // else flip left
  el.style.left = Math.max(8, x) + "px"; el.style.top = r.top + "px";
  const sub = { el, items:[...el.querySelectorAll(".ctx-item")], active:-1, parent:itemEl };
  stack.push(sub); wire(sub); return sub;
}
// ArrowRight → openSub(rec, cur) then setActive(sub, 0)
// ArrowLeft  → closeTop() (only when rec.parent exists)
    
      
    7. 
      
  7. 
        
    7
    Dismiss: outside-click, Esc, scroll, blur
    
        
    A menu must be easy to make go away. Close everything on a pointer-down that lands outside any menu, on window scroll (the anchor moved), and on window blur (you alt-tabbed). Esc closes one level at a time. Use capture-phase listeners so they fire before anything else.
    
        
    document.addEventListener("pointerdown", e => {
  if (stack.length && !e.target.closest(".ctx-menu")) closeAll();
}, true);
addEventListener("scroll", () => stack.length && closeAll(), true);
addEventListener("blur",   () => stack.length && closeAll());
addEventListener("resize", () => stack.length && closeAll());
function closeAll(){ stack.forEach(r => r.el.remove()); stack = []; }
    
      
    8. 
      
  8. 
        
    8
    Activate — and long-press for touch
    
        
    Activating an item runs its action and closes the menu — unless it is disabled (ignore) or has a submenu (open it instead). Touch devices have no right-click, so map a ~500ms long-press to the same openRoot and cancel it if the finger moves.
    
        
    function activate(rec, el){
  const it = el && el._item; if (!it || it.disabled) return;
  if (it.submenu) return openSub(rec, el);   // open, don't run
  run(it); closeAll();
}
let t; surface.addEventListener("touchstart", e => {
  const p = e.touches[0];
  t = setTimeout(() => openRoot(p.clientX, p.clientY), 500);   // long-press
}, { passive:true });
surface.addEventListener("touchmove", () => clearTimeout(t), { passive:true });
surface.addEventListener("touchend",  () => clearTimeout(t));
