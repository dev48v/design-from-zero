🎠 Carousel — Slides, Dots, Swipe & Autoplay

One wide row of slides, slid left and right.

Every slide lives in a single flex row that is N slides wide. To show slide N we just push the row sideways with translateX(-N×100%) and let one CSS transition glide it into place. Use the arrows or the dots, drag/swipe the slide with your mouse or finger, press the ← / → keys, or flip on autoplay to let it advance itself every 3s (it pauses while you hover). It wraps around from the last slide back to the first.

slide 1 / 5
What's happening under the hood

The .carousel is a fixed-width window with overflow:hidden. Inside it the .track is a flex row holding all 5 slides side by side, so it is 5× as wide. Showing slide i is just one line: track.style.transform = "translateX(" + (-i * 100) + "%)", and a CSS transition does the gliding. While you drag, we turn the transition off and follow your finger: translateX(base + dragDistance). On release we measure how far you moved — past ~20% of the width (or a fast flick) snaps to the next/previous slide, otherwise it springs back. The dots and the "slide x / 5" label just mirror the current index.

The whole idea: one flex row of N slides, an index, and translateX(-index×100%). Arrows and dots change the index, drag follows your finger then snaps to the nearest slide, and autoplay nudges the index on a timer — that is the entire carousel.

🔨 Build it yourself

    
A clipped viewport, one flex row of slides, and a single index you move with translateX. Arrows, dots, autoplay, and swipe all just change that index. Pure CSS plus vanilla JS, no libraries.

    
    

      
  1. 1
    The viewport + track markup
    
        
    <!-- the viewport clips to one slide; the track is the wide row -->
<div class="carousel">
  <div class="track">
    <div class="slide">slide 1</div>
    <div class="slide">slide 2</div>
    <div class="slide">slide 3</div>
  </div>
</div>

<!-- or build the slides from a data array: -->
<script>
  track.innerHTML = slides.map(s =>
    `<div class="slide" style="background:${s.bg}">${s.emoji}</div>`
  ).join("");
</script>

    Style: clip the window, lay slides in a flex row
    
        
    .carousel {
  width:520px;
  overflow:hidden;            /* hide everything but one slide */
  position:relative;
}
.track {
  display:flex;               /* slides sit side by side */
  transition:transform .45s ease;
}
.slide {
  flex:0 0 100%;             /* each slide = exactly viewport width */
  height:320px;
}

  3. 3
    goTo(index) — the one line that does the work
    
        
    let index = 0;
const track = document.querySelector(".track");
const count = slides.length;

function goTo(i) {
  index = (i + count) % count;          // wrap negative + overflow
  track.style.transform =
    `translateX(${-index * 100}%)`;     // show slide `index`
  updateDots();                         // keep dots in sync
  status.textContent = `slide ${index + 1} / ${count}`;
}
goTo(0);

  4. 4
    Prev / next arrows + keyboard
    
        
    nextBtn.onclick = () => goTo(index + 1);   // % count wraps to 0
prevBtn.onclick = () => goTo(index - 1);   // % count wraps to last

// arrow keys move the carousel too
document.addEventListener("keydown", e => {
  if (e.key === "ArrowRight") goTo(index + 1);
  if (e.key === "ArrowLeft")  goTo(index - 1);
});
    
      
    5. 

      
  5. 5
    Dot indicators, synced to the index
    
        
    // one dot per slide; clicking a dot jumps to it
dots.innerHTML = slides
  .map((_, i) => `<button class="dot" data-i="${i}"></button>`)
  .join("");
dots.querySelectorAll(".dot").forEach(d =>
  d.onclick = () => goTo(+d.dataset.i)
);

function updateDots() {
  dots.querySelectorAll(".dot").forEach((d, i) =>
    d.classList.toggle("active", i === index)
  );
}

  6. 6
    Autoplay with setInterval + pause on hover
    
        
    let timer = null;
function play()  { stop(); timer = setInterval(() => goTo(index + 1), 3000); }
function stop()  { clearInterval(timer); timer = null; }

// pause while the pointer is over the carousel, resume on leave
carousel.addEventListener("mouseenter", stop);
carousel.addEventListener("mouseleave", () => { if (autoOn) play(); });

autoToggle.onchange = () => { autoOn = autoToggle.checked; autoOn ? play() : stop(); };
    
      
    7. 

      
  7. 7
    Drag / swipe — follow the finger, snap on release
    
        
    let startX = 0, dragging = false, width = 0;

track.addEventListener("pointerdown", e => {
  dragging = true; startX = e.clientX;
  width = carousel.clientWidth;          // px per slide
  track.classList.add("dragging");       // kill the transition
  track.setPointerCapture(e.pointerId);
});

track.addEventListener("pointermove", e => {
  if (!dragging) return;
  const dx = e.clientX - startX;          // how far the finger moved
  // base position for the current slide, plus the live drag (in %)
  const pct = -index * 100 + (dx / width) * 100;
  track.style.transform = `translateX(${pct}%)`;
});
    
      
    8. 

      
  8. 8
    Release: snap to the nearest slide (the threshold)
    
        
    track.addEventListener("pointerup", e => {
  if (!dragging) return;
  dragging = false;
  track.classList.remove("dragging");    // bring the transition back
  const dx = e.clientX - startX;
  // moved past 20% of the width? go that way. else spring back.
  if (dx < -width * 0.2)      goTo(index + 1);   // dragged left → next
  else if (dx > width * 0.2)  goTo(index - 1);   // dragged right → prev
  else                         goTo(index);        // not enough → snap back
});
