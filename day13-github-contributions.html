🔨 Build it yourself

    
A date loop, one CSS Grid, and a count→colour map. ~60 lines, pure CSS Grid + vanilla JS.

    
    

      
  1. 1
    Build a year of dates (back to the first Sunday)
    
        
    // 53 weeks × 7 days. Walk back to the Sunday on/before "one year ago".
const today = new Date();
const start = new Date(today);
start.setDate(start.getDate() - 364);
start.setDate(start.getDate() - start.getDay());  // snap to Sunday

const days = [];
for (let d = new Date(start); d <= today; d.setDate(d.getDate() + 1)) {
  days.push({ date: new Date(d), count: 0 });
}

    Count → level → colour
    
        
    // bucket a raw commit count into 5 levels
function levelFor(count) {
  if (count === 0) return 0;
  if (count <= 3)  return 1;
  if (count <= 6)  return 2;
  if (count <= 9)  return 3;
  return 4;
}
const COLORS = ["#ebedf0","#9be9a8","#40c463","#30a14e","#216e39"];

  4. 4
    Render one cell per day
    
        
    const graph = document.querySelector(".graph");
days.forEach(day => {
  const cell = document.createElement("div");
  cell.className = "cell";
  cell.style.background = COLORS[levelFor(day.count)];
  cell.dataset.date = day.date.toISOString().slice(0, 10);
  graph.appendChild(cell);
});
// grid-auto-flow: column places day 0..6 down column 1, 7..13 down column 2…

    Hover tooltip — "N contributions on date"
    
        
    graph.addEventListener("mouseover", e => {
  const cell = e.target.closest(".cell");
  if (!cell) return;
  const n = cell.dataset.count || 0;
  tooltip.textContent = `${n} contributions on ${cell.dataset.date}`;
  const box = cell.getBoundingClientRect();
  tooltip.style.left = box.left + box.width / 2 + "px";
  tooltip.style.top  = box.top + "px";
  tooltip.classList.add("show");
});
graph.addEventListener("mouseout", () => tooltip.classList.remove("show"));

    Paint on click & drag
    
        
    let painting = false;
function bump(cell) {
  let lvl = (+cell.dataset.level + 1) % 5;   // cycle 0→1→2→3→4→0
  cell.dataset.level = lvl;
  cell.style.background = COLORS[lvl];
}
graph.addEventListener("mousedown", e => {
  const c = e.target.closest(".cell"); if (!c) return;
  painting = true; bump(c); e.preventDefault();
});
graph.addEventListener("mouseover", e => {
  if (!painting) return;
  const c = e.target.closest(".cell"); if (c) bump(c);
});
window.addEventListener("mouseup", () => painting = false);

    A realistic randomizer + scroll on mobile
    
        
    // most days are quiet; weekdays busier; occasional streaks
function randomYear() {
  days.forEach(d => {
    const weekend = [0,6].includes(d.date.getDay());
    let n = Math.random() < (weekend ? 0.35 : 0.7)
          ? Math.floor(Math.random() * (weekend ? 4 : 12)) : 0;
    setCell(d, n);
  });
}
/* wrap the graph in a scroll container so 53 columns scroll on phones */
.scroll { overflow-x: auto; }
