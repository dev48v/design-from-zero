A list in state, a fixed container, and a timeout per toast. ~40 lines.

    
    
      
  1. 1
    Toasts are a list of state
    let toasts = [];   // [{ id, type, message }]
function toast(message, type="info") {
  toasts.push({ id: Date.now()+Math.random(), type, message });
  render();
}

    Auto-dismiss with a timer

function dismiss(id) { toasts = toasts.filter(t => t.id !== id); render(); }
