📸 Instagram Stories — Segmented Progress Bars

Watch the bar fill, then jump to the next story.

Across the top sit one thin bar per slide. Only the current bar animates its width from 0→100% over ~4s; finished bars stay full, upcoming ones stay empty. When it fills, it auto-advances. Tap the right half to skip ahead, the left half to go back, and press-and-hold anywhere to pause the timer (release to resume).

slide 1 / 5
What's happening under the hood

There are N segments — one per story. Each segment is a track with a .seg-fill inside it. Only the active segment animates: a requestAnimationFrame loop nudges its width from 0% to 100% over the chosen duration. When it reaches 100% we advance: the segment snaps full, the next one becomes active and starts at 0%. Past segments are forced to 100%, future ones to 0%. Tapping a half jumps the index; pressing and holding just stops feeding the loop, so the fill freezes until you let go.

The whole idea: a row of N bars where only the active one grows 0→100%; finishing (or a right-tap) advances the index, a left-tap rewinds it, and holding pauses the timer — that's the entire Stories pattern.

  3. 3
    Animate ONLY the active bar with requestAnimationFrame
    
        
    let index = 0, elapsed = 0, last = 0;
const DURATION = 4000;   // ms per slide

function tick(now) {
  if (!last) last = now;
  if (!paused) elapsed += now - last;
  last = now;

  const pct = Math.min(elapsed / DURATION, 1) * 100;
  fills[index].style.width = pct + "%";   // grow the active fill

  if (pct >= 100) next();                  // full → advance
  raf = requestAnimationFrame(tick);
}
requestAnimationFrame(tick);
    
      
    4. 

      
  4. 4
    Advance, rewind, and reset each segment's state
    
        
    function paint() {
  fills.forEach((f, i) => {
    if (i < index)      f.style.width = "100%"; // past = full
    else if (i > index) f.style.width = "0%";   // future = empty
    // the active one is left for the rAF loop to fill
  });
  showSlide(index);
}

function next() {
  if (index < slides.length - 1) { index++; elapsed = 0; }
  else if (loop)                 { index = 0; elapsed = 0; }
  paint();
}
function prev() { index = Math.max(0, index - 1); elapsed = 0; paint(); }

  6. 6
    Press-and-hold to pause (pointerdown / pointerup)
    
        
    let paused = false;

// any press pauses; release resumes. Use pointer events so it
// works for mouse AND touch with one set of handlers.
phone.addEventListener("pointerdown", () => paused = true);
phone.addEventListener("pointerup",   () => { paused = false; last = 0; });
phone.addEventListener("pointerleave",() => { paused = false; last = 0; });

// resetting `last = 0` on resume stops the paused gap from being
// counted as elapsed time on the very next frame.

    Swap the slide and preload the next one
    
        
    function showSlide(i) {
  stage.style.background = slides[i].bg;   // gradient / image
  stage.textContent      = slides[i].emoji;
  // preload the NEXT image so it's ready before we advance
  const n = slides[i + 1];
  if (n && n.img) { const im = new Image(); im.src = n.img; }
}

  8. 8
    Loop or end — and stay accessible
    
        
    // at the last slide: loop back, or stop and show an "end" state
function next() {
  if (index < slides.length - 1) index++;
  else if (loop)                 index = 0;
  else { cancelAnimationFrame(raf); onEnd(); return; }
  elapsed = 0; paint();
}

// accessibility: never trap the user — pausing on hold is the
// escape hatch, so a slow reader can stop the auto-advance.
// also expose buttons (prev/next) so it works without taps.
