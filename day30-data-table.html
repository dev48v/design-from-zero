Sort, resize, search and page through a real table.

A working data table over 44 employees, built from scratch in vanilla JS — no grid library. Click a header to sort (it cycles ascending → descending → off, and it is type-aware: salary sorts as numbers, the join date sorts chronologically, names alphabetically). 🔨 Build it yourself

    
A table described as data, rendered through a colgroup so widths are drag-adjustable, sorted by three type-aware comparators in a tri-state cycle, filtered live with a debounce, paged with a single slice(), and pinned with a sticky header — every interaction funnelled through one derive pipeline so sort + filter + page never fight. ~150 lines of vanilla JS, no dependencies.

    
    
      
  1. 
        
    1
    Describe the data and the columns
    
        
    Rows are a plain array of objects. Columns are their own array of descriptors — a key into each row, a label, a type (that drives the sort), a starting width, and an optional format for display. Keep the raw value for sorting; format only when you paint the cell.
    
        
    const ROWS = [
  { name:"Ava Chen", role:"Frontend Engineer", dept:"Engineering",
    salary:118000, joined:"2021-03-14" },
  /* …43 more… */
];
const COLUMNS = [
  { key:"name",   label:"Name",       type:"string", width:200 },
  { key:"role",   label:"Role",       type:"string", width:210 },
  { key:"dept",   label:"Department", type:"string", width:150 },
  { key:"salary", label:"Salary",     type:"number", width:130, align:"right",
    format:v => "$" + v.toLocaleString() },
  { key:"joined", label:"Joined",     type:"date",   width:140,
    format:v => fmtDate(v) },
];

  2. 
        
    2
    Render from data — with a colgroup
    
        
    Never hand-write <tr>s. Build a <colgroup> with one <col> per column (this is what makes widths controllable), set table-layout:fixed, then loop the rows into the body. Re-rendering the body is now just clearing tbody and looping the current page.
    
        
    // one <col> per column → the single source of truth for width
COLUMNS.forEach(c => {
  const col = document.createElement("col");
  col.style.width = c.width + "px";
  colgroup.appendChild(col);
});
// body: clear + loop (never hardcoded)
tbody.innerHTML = "";
pageRows.forEach(r => {
  const tr = document.createElement("tr");
  COLUMNS.forEach(c => {
    const td = document.createElement("td");
    if (c.align === "right") td.classList.add("ta-right");
    td.textContent = c.format ? c.format(r[c.key]) : r[c.key];
    tr.appendChild(td);
  });
  tbody.appendChild(tr);
});

  3. 
        
    3
    Type-aware comparators + tri-state sort
    
        
    The classic bug: [9, 80, 100].sort() gives [100, 80, 9] because the default sort compares strings. So branch on the column's type — numbers subtract, dates subtract as Date objects, strings use localeCompare. Clicking a header cycles none → ascending → descending → none; multiply the comparator by the direction.
    
        
    function compare(a, b, type){
  if (type === "number") return a - b;                 // numeric, not lexical
  if (type === "date")   return new Date(a) - new Date(b);  // chronological
  return String(a).localeCompare(String(b));           // alphabetical
}
function cycleSort(key){
  if (state.sortKey !== key){ state.sortKey = key; state.sortDir = 1; }   // asc
  else if (state.sortDir === 1) state.sortDir = -1;                        // desc
  else if (state.sortDir === -1){ state.sortDir = 0; state.sortKey = null; } // off
  else state.sortDir = 1;
  render();
}

  4. 
        
    4
    Drag to resize — pointer capture + min-width
    
        
    Drop a thin handle on each header's right edge. On pointerdown record the start X and the column's start width and setPointerCapture so you keep getting moves even if the cursor outruns the handle. On pointermove set the matching <col>'s width to startW + dx, clamped to a minimum so a column can't vanish. Stop the event so the same click doesn't trigger a sort.
    
        
    function startResize(e, i, colEl){
  e.preventDefault(); e.stopPropagation();      // don't let it become a sort
  const startX = e.clientX, startW = colEl.getBoundingClientRect().width;
  const h = e.currentTarget; h.setPointerCapture(e.pointerId);
  function move(ev){
    const w = Math.max(MIN_W, startW + (ev.clientX - startX));  // never below min
    COLUMNS[i].width = w; colEl.style.width = w + "px";         // drive the <col>
  }
  function up(){ h.releasePointerCapture(e.pointerId);
                 h.removeEventListener("pointermove", move);
                 h.removeEventListener("pointerup", up); }
  h.addEventListener("pointermove", move);
  h.addEventListener("pointerup", up);
}

  5. 
        
    5
    Live search — with a debounce
    
        
    Filtering keeps rows where any column contains the query (lower-cased both sides). But re-rendering on every keystroke is wasteful, so debounce: wait ~150ms after the last key before you filter. Filtering changes the row count, so reset to page 1 — otherwise you can be stranded on page 5 of a 2-page result.
    
        
    let deb;
search.addEventListener("input", e => {
  clearTimeout(deb);
  deb = setTimeout(() => {
    state.filter = e.target.value.trim().toLowerCase();
    state.page = 1;                 // filtering resets paging
    render();
  }, 150);                          // wait for the typing to settle
});
// inside the pipeline:
rows = rows.filter(r =>
  COLUMNS.some(c => String(r[c.key]).toLowerCase().includes(state.filter)));
    
      
    6. 
      
  6. 
        
    6
    Pagination — one slice, careful math
    
        
    A page is just a window into the array. Total pages is ceil(total / size). Clamp the current page into range (so shrinking the result never leaves you past the end), then take slice(start, start + size) where start = (page − 1) × size. That single slice is the only thing the body ever renders.
    
        
    const total = rows.length;
const pages = Math.max(1, Math.ceil(total / state.pageSize));
state.page   = Math.min(Math.max(1, state.page), pages);   // clamp into range
const start  = (state.page - 1) * state.pageSize;
const pageRows = rows.slice(start, start + state.pageSize); // the window
// footer readout: e.g. "13–24 of 44"
range.textContent = (total ? start + 1 : 0) + "–" +
                    (start + pageRows.length) + " of " + total;

    Sticky header inside a scroll container
    
        
    Wrap the table in a box with a fixed max-height and overflow:auto — that box is what scrolls, not the page. Then position:sticky; top:0 on the header cells pins them to the top of that box. Give the header a solid background and a higher z-index so scrolling rows pass cleanly underneath it.
    
        
    .dt-wrap { max-height: 460px; overflow: auto; }   /* the scroll box */

.dt thead th {
  position: sticky; top: 0;      /* pin to the top of .dt-wrap */
  z-index: 2;                    /* above the body rows */
  background: #f8fafc;           /* opaque, or rows show through */
}

    8
    One pipeline + accessibility
    
        
    Keep sort, filter and page in one state object and derive the view in a fixed order every render: filter → sort → paginate. Never mutate the source array (sort a copy). For screen readers, headers are <th scope="col"> and the sorted one carries aria-sort="ascending"/"descending", kept off the others.
    
        
    function getView(){
  let rows = ROWS;                                       // never mutate source
  if (state.filter) rows = rows.filter(matchesFilter);   // 1. filter
  if (state.sortKey && state.sortDir)
    rows = rows.slice().sort((a, b) =>                   // 2. sort (a copy)
      compare(a[state.sortKey], b[state.sortKey], type) * state.sortDir);
  return paginate(rows);                                 // 3. paginate
}
// a11y: reflect the sort on the header
th.setAttribute("scope", "col");
if (sorted) th.setAttribute("aria-sort", dir === 1 ? "ascending" : "descending");
else        th.removeAttribute("aria-sort");
