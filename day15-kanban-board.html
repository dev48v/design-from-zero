🗂️ Kanban Board — Drag-and-Drop Columns

Drag a card. Watch work move through stages.

Grab any card and drop it into another column. The column you hover lights up, a dashed gap shows where it'll land, and the card counts in every header update instantly. Add cards with the input at the bottom of a column, or hit × to delete. All native HTML5 drag-and-drop — no library.

What's happening under the hood

On dragstart we stash the card's id in dataTransfer. A column is only a valid drop zone if its dragover handler calls preventDefault() — the classic gotcha, since by default nothing accepts a drop. On drop we read the id back, move that card's DOM node into the column, update our model, and re-count each header.

The whole trick: the browser fires dragstart → dragover → drop. You carry the dragged item's id in dataTransfer, you must preventDefault() on dragover to allow the drop, and on drop you move the node + update state. Counts and persistence are just bookkeeping on top.

    The markup — three columns, each a drop zone

    <header>To Do <span class="count">0</span></header>

  2. 2
    Keep a model, then render from it
    
        
    // the model is the single source of truth — id → text + column
let cards = [
  { id:"c1", text:"Design the empty state", col:"todo"  },
  { id:"c2", text:"Wire dragover handler",  col:"doing" },
  { id:"c3", text:"Ship v1",                col:"done"  },
];
let nextId = 4;

// render() wipes each list and rebuilds cards from the model
function render() {
  document.querySelectorAll(".list").forEach(l => l.innerHTML = "");
  cards.forEach(c => columnList(c.col).appendChild(cardEl(c)));
  updateCounts();
}
    
      
    3. 

      
  3. 3
    Make a card draggable + fire dragstart
    
        
    function cardEl(c) {
  const el = document.createElement("div");
  el.className = "card";
  el.draggable = true;            // THIS makes it grabbable
  el.dataset.id = c.id;
  el.textContent = c.text;

  // dragstart: stash WHICH card is moving, in dataTransfer
  el.addEventListener("dragstart", e => {
    e.dataTransfer.setData("text/plain", c.id);
    e.dataTransfer.effectAllowed = "move";
    el.classList.add("dragging");
  });
  el.addEventListener("dragend", () => el.classList.remove("dragging"));
  return el;
}
    
      
    4. 

      
  4. 4
    The gotcha: preventDefault on dragover
    
        
    // By default NOTHING is a drop target. You must cancel
// dragover (and dragenter) for the browser to allow a drop.
document.querySelectorAll(".col").forEach(col => {
  col.addEventListener("dragover", e => {
    e.preventDefault();                 // <-- without this, no drop fires
    e.dataTransfer.dropEffect = "move";
    col.classList.add("drop-target");   // highlight the zone
  });
  col.addEventListener("dragleave", () => col.classList.remove("drop-target"));
});
    
      
    5. 

      
  5. 5
    On drop: read the id, update the model
    
        
    col.addEventListener("drop", e => {
  e.preventDefault();
  col.classList.remove("drop-target");

  const id  = e.dataTransfer.getData("text/plain");  // who moved
  const dst = col.dataset.col;                        // where to

  // update the MODEL first, then re-render the DOM from it
  const card = cards.find(c => c.id === id);
  if (card) card.col = dst;
  render();
  save();
});
    
      
    6. 

      
  6. 6
    Live counts in each header
    
        
    function updateCounts() {
  document.querySelectorAll(".col").forEach(col => {
    const id = col.dataset.col;
    const n  = cards.filter(c => c.col === id).length;
    col.querySelector(".count").textContent = n;
  });
}
// counts are derived from the model, so they're always correct
// after any move, add, or delete — no manual ++ / -- needed.
    
      
    7. 

      
  7. 7
    Add a card + delete a card
    
        
    function addCard(colId, text) {
  if (!text.trim()) return;
  cards.push({ id:"c" + (nextId++), text:text.trim(), col:colId });
  render(); save();
}
function deleteCard(id) {
  cards = cards.filter(c => c.id !== id);   // drop it from the model
  render(); save();
}
// each card gets a × button wired to deleteCard(c.id)
    
      
    8. 

      
  8. 8
    Persist to localStorage
    
        
    function save() {
  localStorage.setItem("kanban", JSON.stringify(cards));
}
function load() {
  const saved = localStorage.getItem("kanban");
  if (saved) cards = JSON.parse(saved);
}
// load() on startup, save() after every change → the board
// survives a refresh. Reset = localStorage.removeItem("kanban").
load(); render();
